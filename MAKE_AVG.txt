package com.mycompany.makeavg;
import java.util.*;
public class MAKEAVG {

    public static void main(String[] args) {
       Scanner input = new Scanner(System.in);
        int A = input.nextInt();
        int C = input.nextInt();

        if ((A + C) % 2 == 0) {
            System.out.println("YES");
        } else {
            System.out.println("NO");
        }
    }
}
