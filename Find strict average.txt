package com.mycompany.find_strict_average;
import java.util.*;
public class Find_Strict_Average {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int A = input.nextInt();
        int B = input.nextInt();
        int C = input.nextInt();
        if ((A + B) / 2.0 > C) {
            System.out.println("YES");
        } else {
            System.out.println("NO");
        }
    }
}
