package javacodes5;
import java.util.*;

/**
 *
 * @author USER
 */
public class Javacodes5 {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter a number: "); 
        int N = input.nextInt();

        for (int i = 1; i <= N; i++) {
            System.out.print(i + " ");
        }
    }
}
