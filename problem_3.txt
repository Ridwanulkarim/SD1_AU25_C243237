package javacodes;
import java.util.*;


/**
 *
 * @author USER
 */
public class JavaCodes {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter a number:");
        int a = input.nextInt();
        int b = input.nextInt();
        int c = input.nextInt();
        int max = a;
        if(b > max) max = b;
        if(c > max) max = c;
        System.out.println(max);
    }
}
