package com.mycompany.javacodes9;
import java.util.*;
/**
 *
 * @author USER
 */
public class Javacodes9 {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int n = input.nextInt();

        int fact = 1;
        for (int i = 1; i <= n; i++) {
            fact *= i;
        }
        System.out.println("Factorial = " + fact);
    }
}
